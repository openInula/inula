11. 当前不支持for (let key in numbers)这种写法
```js
it('should observe enumeration', () => {
  let dummy = 0;
  const numbers = reactive<Record<string, number>>({ num1: 3 });
  effect(() => {
    dummy = 0;
    for (let key in numbers) {
      dummy += numbers[key];
    }
  });

  expect(dummy).toBe(3);
  numbers.num2 = 4;
  expect(dummy).toBe(7);
  delete numbers.num1;
  expect(dummy).toBe(4);
});
```
